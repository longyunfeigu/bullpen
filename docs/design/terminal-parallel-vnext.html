/** Same xterm instance moves between dock and side focus slot. */
export function mountTerminal(host: HTMLElement, item: TermInstance) {
const element = item.term.element;
if (!element) item.term.open(host);
else if (element.parentElement !== host) host.replaceChildren(element);
item.fit.fit();
item.term.refresh(0, item.term.rows - 1);
item.term.focus();
}
export function promote(terminalId: string) {
const context = terminalContexts.get(terminalId);
sideSlot.swap({ terminalId, context });
}
